\usepackage{ifthen}
\newcommand{\citeurl}[5]{%
#1\footnote{\emph{#2}%
          \ifthenelse{\equal{#3}{}}%
                     {}%          then i.e. empty #3
                     {, #3}% else i.e. non-empty #3
          \ifthenelse{\equal{#4}{}}%
                     {}%          then i.e. empty #4
                     {, access date: #4}% else i.e. non-empty #4
, \url{#5}}}
